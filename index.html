  <script>

    let asistentes = JSON.parse(localStorage.getItem("asistentes")) || [];

    let contador = asistentes.length + 1;


    function actualizarHora() {

      const ahora = new Date();

      const hora = ahora.toLocaleTimeString();

      document.getElementById("hora").value = hora;

    }


    setInterval(actualizarHora, 1000);


    function guardarAsistentes() {

      localStorage.setItem("asistentes", JSON.stringify(asistentes));

    }


    function renderizarTabla() {

      const tbody = document.querySelector("#tablaAsistencia tbody");

      tbody.innerHTML = "";


      asistentes.forEach((asistente, index) => {

        const fila = document.createElement("tr");


        fila.innerHTML = `

          <td>${index + 1}</td>

          <td>${asistente.nombre}</td>

          <td>${asistente.cedula}</td>

          <td>${asistente.cargo}</td>

          <td><input type="checkbox" ${asistente.asistencia ? "checked" : ""} onchange="marcarAsistencia(${index}, this.checked)"></td>

        `;


        tbody.appendChild(fila);

      });

    }


    function agregarAsistente() {

      const nombre = document.getElementById("nombre").value.trim();

      const cedula = document.getElementById("cedula").value.trim();

      const cargo = document.getElementById("cargo").value.trim();


      if (!nombre || !cedula || !cargo) {

        alert("Por favor, completa todos los campos.");

        return;

      }


      const nuevo = {

        nombre,

        cedula,

        cargo,

        asistencia: false

      };


      asistentes.push(nuevo);

      guardarAsistentes();

      renderizarTabla();


      contador++;

      document.getElementById("nombre").value = "";

      document.getElementById("cedula").value = "";

      document.getElementById("cargo").value = "";

    }


    function marcarAsistencia(index, estado) {

      asistentes[index].asistencia = estado;

      guardarAsistentes();

    }


    // Fecha actual por defecto

    document.getElementById("fecha").valueAsDate = new Date();

    actualizarHora();

    renderizarTabla();

  </script>
